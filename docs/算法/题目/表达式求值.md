表达式求值有三个地方需要考虑。

1. 括号
2. 负数
3. 操作符

目前了解到表达式求值有两种解法，一种是双栈法，一种是递归法。

## 1. 括号的处理

### 双栈法

双栈法中把括号当作操作符，遇到左括号直接入栈，遇到右括号之后，右括号不入栈，出栈一个操作符和两个操作数，进行运算，然后把计算结果入操作数栈。直到遇到左括号出栈。

```c++
while(!op_stk.empty() && op_stack.top()!='('){
  int b = num_stk.top(); // 先出栈的是第二个操作数，考虑减法或除法
  num_stk.pop();
  int a = num_stk.top();
  num_stk.pop();
  char op = op_stk.top();
  op_stk.pop();
  num_stk.push(calc(a,b,op));
}
op_stack.pop(); // 左括号出栈
```

### 递归法

递归法中，先找到表达式中优先级最小的操作符，然后分别递归获取操作符两边的值，然后进行运算。
所以我们在递归法中处理括号的时候维护一个括号优先级变量，在计算操作符优先级的时候加上括号优先级。

```c++
int min_priority_operation_index = -1;
int temp_priority = 0, current_priority = INF, min_priority=INF;

for(int i=left;i<right;i++){
  if(!is_operation(s[i])) continue;
  switch(s[i]){
    case '(':
      temp_priority += 100;
      break;
    case ')':
      temp_priority -= 100;
      break;
    case '+':
    case '-':
      current_priority = 1 + temp_priority;
      break;
    case '*':
    case '/':
      current_priority = 2 + temp_priority;
    break;
  }

  if((s[i]=='-' || s[i]=='+') && (i==left || is_operation(s[i-1]) || s[i-1] == '(')){
    current_priority += 1000; // 兼容负号，负号优先级最高，这样在当前这段表达式中除非只是一个负数，否则不会被处理。
  }

  if(current_priority < min_priority){
    min_priority = current_priority;
    min_priority_operation_index = i;
  }
}

```

## 2. 负数的处理

判断“-”是减号还是负号

```c++
if(s[i] == '-'){
  if(i==0 || s[i-1] == '(' || is_operation(s[i-1])){
    // 负号
  }
  else{
    // 减号
  }
}
```

在主流程中，可以先处理数值，在处理数值的时候同时处理符号，这样处理操作符的时候遇到的就肯定是减号了。

```c++
  vector<string> stringToInfix(string s){
    vector<string> res;
    for(int i=0;s[i];i++){
      // 处理数字，兼容负数
      if((s[i]>='0' && s[i]<='9') || ((s[i]=='-' || s[i]=='+')&&i==0 || s[i-1]=='(' || is_operation(s[i-1]))){
        int j=i+1;
        while(s[j]>='0' && s[j]<='9')j++;
        res.push_back(s.substr(i,j-i));
        i=j-1;
        continue;
      }

      if(is_operation(s[i])){
        res.push_back(s.substr(i,1));
      }
    }
    return res;
  }
```

## 3. 操作符的处理

### 双栈法

在双栈法中，遇到操作符，我们需要入栈当前操作符，但是在入栈前需要先把前面入栈的高优先级的操作符出栈进行计算，直到遇到低优先级的操作符或者遇到左括号。

```c++
while(!op_stk.empty() && op_stk.top()!='(' && get_priority(op_stk.top())>=get_priority(s[i])){
  int b = num_stk.top();
  num_stk.pop();
  int a = num_stk.top();
  num_stk.pop();
  char op = op_stk.top();
  op_stk.pop();
  num_stk.push(calc(a,b,op));
}
op_stk.push(s[i]);
```

### 转为后缀表达式

遇到数值的时候直接追加到后缀表达式中，遇到操作符的时候类似于双栈法，先把操作符放到一个临时的栈中，然后把临时栈中优先级大于等于当前操作符的操作符拿出放到后缀表达式中。

```c++
void infixToPostfix(const vector<string> &s, vector<string> &postfix) {
  stack<string> op_stk;
  for (string x : s) {
    if (is_num(x)) {
      postfix.push_back(x);
    } else {
      if (x == "(") {
        op_stk.push(x);
      } else if (x == ")") {
        while (!op_stk.empty() && op_stk.top() != "(") {
          postfix.push_back(op_stk.top());
          op_stk.pop();
        }
        op_stk.pop(); // pop (;
      } else {
        while (!op_stk.empty() && priority(op_stk.top()[0]) >= priority(x[0])) {
          postfix.push_back(op_stk.top());
          op_stk.pop();
        }
        op_stk.push(x);
      }
    }
  }

  while (!op_stk.empty()) {
    if (op_stk.top() != "(" && op_stk.top() != ")")
      postfix.push_back(op_stk.top());
    op_stk.pop();
  }
}
```

```c++
#include <cmath>
#include <iostream>
#include <string>

using namespace std;

typedef long long ll;

bool is_operator(char c) {
  return c == '(' || c == '+' || c == '-' || c == '*' || c == '/' || c == '^';
}
ll calc(string s, ll l, ll r) {
#define INF 0x7f7f7f7f
  ll pos = -1, min_pri = INF - 1, cur_pri = INF, temp_pri = 0;
  for (int i = l; i < r; i++) {
    switch (s[i]) {
    case '(':
      temp_pri += 100;
      break;
    case ')':
      temp_pri -= 100;
      break;
    case '+':
      cur_pri = 1 + temp_pri;
      break;
    case '-':
      cur_pri = 1 + temp_pri;
      break;
    case '*':
      cur_pri = 2 + temp_pri;
      break;
    case '/':
      cur_pri = 2 + temp_pri;
      break;
    case '^':
      cur_pri = 3 + temp_pri;
      break;
    }

    if ((s[i] == '+' || s[i] == '-') && (i == 0 || is_operator(s[i - 1]))) {
      cur_pri += 1000;
    }
    if (cur_pri <= min_pri) {
      min_pri = cur_pri;
      pos = i;
      cur_pri = INF;
    }
  }

  if (pos == -1) {
    ll res = 0;
    for (int i = l; i < r; i++) {
      if (s[i] < '0' || s[i] > '9')
        continue;
      res = res * 10 + (s[i] - '0');
    }
    return res;
  }

  ll a = calc(s, l, pos);
  ll b = calc(s, pos + 1, r);
  switch (s[pos]) {
  case '+':
    return a + b;
    break;
  case '-':
    return a - b;
    break;
  case '*':
    return a * b;
    break;
  case '/':
    return a / b;
    break;
  case '^':
    return pow(a, b);
    break;
  }

  return 0;
}

int main() {
  string expr;
  cin >> expr;
  ll res = calc(expr, 0, expr.size());
  cout << res << endl;
  return 0;
}
```

## 双栈法

```c++
#include <iostream>
#include <stack>
#include <string>

using namespace std;

int getNum(const string &s, int &i) {
  int j = i + 1;
  while (s[j] >= '0' && s[j] <= '9')
    j++;
  string res = s.substr(i, j - i);
  i = j - 1;
  cout << "res: " << res << endl;
  return stoi(res);
}

int priority(const char op) {
  switch (op) {
  case '+':
    return 1;
  case '-':
    return 1;
  case '*':
    return 2;
  case '/':
    return 2;
  }
  return 0;
}

bool is_op(const char op) { return priority(op) > 0; }

int calc(int a, int b, char op) {
  switch (op) {
  case '+':
    return a + b;
  case '-':
    return a - b;
  case '*':
    return a * b;
  case '/':
    return a / b;
  }
  return 0;
}

int getRes(string s) {
  stack<int> nums;
  stack<char> ops;
  for (int i = 0; s[i]; i++) {
    if (s[i] == '(') {
      ops.push(s[i]);
      continue;
    }

    if (s[i] == ')') {
      while (!ops.empty() && priority(ops.top()) >= priority(s[i])) {
        if (ops.top() == '(') {
          ops.pop();
          break;
        }
        int b = nums.top();
        nums.pop();
        int a = nums.top();
        nums.pop();
        nums.push(calc(a, b, ops.top()));
        ops.pop();
      }
      continue;
    }

    if ((s[i] == '+' || s[i] == '-') &&
        (i == 0 || s[i - 1] == '(' || is_op(s[i - 1]))) {
      int num = getNum(s, i);
      nums.push(num);
      continue;
    }
    if (is_op(s[i])) {
      while (!ops.empty() && priority(ops.top()) >= priority(s[i])) {
        int b = nums.top();
        nums.pop();
        int a = nums.top();
        nums.pop();
        nums.push(calc(a, b, ops.top()));
        ops.pop();
      }
      ops.push(s[i]);
      continue;
    }
    int num = getNum(s, i);
    nums.push(num);
  }
  while (!ops.empty()) {
    int b = nums.top();
    nums.pop();
    int a = nums.top();
    nums.pop();
    nums.push(calc(a, b, ops.top()));
    ops.pop();
  }
  return nums.top();
}

int main() {
  string expression;
  cin >> expression;
  int res = getRes(expression);
  cout << res << endl;
  return 0;
}
```
