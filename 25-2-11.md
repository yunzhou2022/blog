# 中间件的创建历程

- 为了方便我们更加清晰的了解store当中state的每一步变更，我们需要在每次派发action的时候都可以通过console.log打印出相关信息。

```jsx
function addLogToDispatch(store){
	const rawDispatch = store.dispatch;
	return function dispatch(action){
		cosnole.log("prev state", store.getState());
		console.log(action);
		const res = rawDispatch(action);
		console.log("after state", store.getState());
		return res;
	}
}
```

- 我们发现在调用addPromiseToDispatch的时候，内部的rawDispatch命名不太合适，因为在这里的dispatch是经过addLogToDispatch增强过的， 同理addLogToDispatch中的rawDispatch命名也不太合适，因此我们将命名修改为next。

```jsx
function addLogToDispatch(store){
	const next = store.dispatch;
	return function(action){
		// ...
		return next(action);
	}
}

function addPromiseToDispatch(store){
	const next = store.dispatch;
	return function(action){
		if(typeof action.then === 'function'){
			return action.then(next)
		}
		return next(action);
	}
}
```

- 在实际开发中，如果由开发者手动处理各种改写逻辑，会非常混乱，为了避免这种反模式，我们将这个包装过程收敛—声明一个数据，即中间件数据middlewares

```jsx
const configureStore = ()=>{
	const store = createStore();
	const middlewares = [];
	middlewares.push(addLogToDispatch);
	middlewares.push(addPromiseToDispatch);
  wrapDispatchWithMiddlewares(store, middlewares);

  return store;
}

function wrapDispatchWithMiddlewares(store, middlewares){
	middlewares.forEach(middleware=>{
		store.dispatch = middleware(store)(store.dispatch);
	})
}
```

1. redux源码

```jsx
export default function applyMiddleware(
  ...middlewares
) {
  return createStore => (reducer, preloadedState) => {
    const store = createStore(reducer, preloadedState)
    let dispatch = () => {
      throw new Error(
        'Dispatching while constructing your middleware is not allowed. ' +
          'Other middleware would not be applied to this dispatch.'
      )
    }

    const middlewareAPI = {
      getState: store.getState,
      dispatch: (action, ...args) => dispatch(action, ...args)
    }
    const chain = middlewares.map(middleware => middleware(middlewareAPI))
    dispatch = compose(...chain)(store.dispatch)

    return {
      ...store,
      dispatch
    }
  }
}

export default function compose(...funcs) {
  if (funcs.length === 0) {
    // infer the argument type so it is usable in inference down the line
    return (arg) => arg
  }

  if (funcs.length === 1) {
    return funcs[0]
  }

  return funcs.reduce(
    (a, b) =>
      (...args) =>
        a(b(...args))
  )
}
```
